kUrlObfushortner

Compress long URLs entirely client-side using DEFLATE + URL-safe Base64. No server, no database. The compressed token is the data — decode it anywhere you have JavaScript, PHP, Python, or C#. Pair with kUrlObfushortner-secure to encrypt tokens with RSA-OAEP so only your server can decode them.

How it works

01

Take a long URL

Any URL with a hefty query string — analytics params, filter state, coordinates, whatever.

02

DEFLATE compress

The query string is compressed with DEFLATE (level 9) via pako. Query strings are highly repetitive — they compress well.

03

URL-safe Base64

The binary output is encoded to URL-safe Base64 (-_ instead of +/, no padding) and placed in a single ?_k= param.

04

Decode anywhere

Any receiver that knows the scheme can reverse it: Base64-decode, DEFLATE-inflate, restore the query string. No server lookup needed.

⚠ Compression only pays off for longer query strings (>~80 chars). Short URLs may actually grow slightly after encoding — the stats panels will tell you exactly how much you're saving (or spending).

PHP 8+

Encode & Decode

<?php
// Requires the php-zlib extension (enabled by default)

function kurlEncode(string $url): string {
  $parsed = parse_url($url);
  $qs = $parsed['query'] ?? '';
  if (!$qs) return $url;

  // deflate-raw (wbits = -15)
  $compressed = gzdeflate($qs, 9);
  $token = '1' . rtrim(
    strtr(base64_encode($compressed), '+/', '-_'), '='
  );
  $base = $parsed['scheme'] . '://'
        . $parsed['host']
        . ($parsed['path'] ?? '/');
  return $base . '?_k=' . rawurlencode($token);
}

function kurlDecode(string $url): string {
  parse_str(parse_url($url, PHP_URL_QUERY), $params);
  $token = $params['_k'] ?? '';
  if (!$token) return $url;

  $payload = substr($token, 1); // strip version
  $pad = str_repeat('=', (4 - strlen($payload) % 4) % 4);
  $b64 = strtr($payload . $pad, '-_', '+/');
  $qs  = gzinflate(base64_decode($b64));
  $parsed = parse_url($url);
  $base = $parsed['scheme'] . '://'
        . $parsed['host']
        . ($parsed['path'] ?? '/');
  return $base . '?' . $qs;
}
Python 3

Encode & Decode

import zlib, base64
from urllib.parse import urlparse, urlencode, quote

def kurl_encode(url: str) -> str:
    p = urlparse(url)
    qs = p.query
    if not qs:
        return url
    compressed = zlib.compress(qs.encode(), 9)[2:-4]  # raw deflate
    b64 = base64.urlsafe_b64encode(compressed).rstrip(b'=').decode()
    token = '1' + b64
    base = f"{p.scheme}://{p.netloc}{p.path}"
    return f"{base}?_k={quote(token)}"

def kurl_decode(url: str) -> str:
    p = urlparse(url)
    params = dict(pair.split('=', 1)
               for pair in p.query.split('&') if '=' in pair)
    token = params.get('_k', '')
    if not token:
        return url
    payload = token[1:]  # strip version char
    pad = '=' * ((4 - len(payload) % 4) % 4)
    compressed = base64.urlsafe_b64decode(payload + pad)
    # wrap as zlib stream for inflate
    qs = zlib.decompress(compressed, -15).decode()
    base = f"{p.scheme}://{p.netloc}{p.path}"
    return f"{base}?{qs}"
C# / .NET 6+

Encode & Decode

using System.IO.Compression;
using System.Text;

public static class KUrlObfushortner
{
  public static string Encode(string url)
  {
    var uri = new Uri(url);
    var qs = uri.Query; // includes leading "?"
    if (string.IsNullOrEmpty(qs)) return url;

    var bytes = Encoding.UTF8.GetBytes(qs);
    using var ms = new MemoryStream();
    using (var ds = new DeflateStream(ms, CompressionLevel.SmallestSize))
      ds.Write(bytes, 0, bytes.Length);

    var token = "1" + Convert
      .ToBase64String(ms.ToArray())
      .Replace('+', '-').Replace('/', '_')
      .TrimEnd('=');

    var base_ = $"{uri.Scheme}://{uri.Authority}{uri.AbsolutePath}";
    return $"{base_}?_k={Uri.EscapeDataString(token)}";
  }

  public static string Decode(string url)
  {
    var uri   = new Uri(url);
    var query = System.Web.HttpUtility.ParseQueryString(uri.Query);
    var token = query["_k"];
    if (token == null) return url;

    var payload = token[1..]; // strip version
    var pad = new string('=', (4 - payload.Length % 4) % 4);
    var b64 = payload.Replace('-', '+').Replace('_', '/') + pad;
    var compressed = Convert.FromBase64String(b64);

    using var ms = new MemoryStream(compressed);
    using var ds = new DeflateStream(ms, CompressionMode.Decompress);
    using var sr = new StreamReader(ds, Encoding.UTF8);
    var qs = sr.ReadToEnd();

    var base_ = $"{uri.Scheme}://{uri.Authority}{uri.AbsolutePath}";
    return base_ + qs;
  }
}

Limitations & Roadmap (Base Plugin)

Current limitations of the base plugin:

Compressed, not encrypted

The base token is DEFLATE + Base64 only. Anyone who knows the scheme can decode it. Use kUrlObfushortner-secure for confidentiality.

Server must cooperate

The receiving server must recognise _k and decode it (or serve the JS that does). This demo page acts as its own receiver.

Short URLs may grow

Very short query strings (<~80 chars) may expand slightly due to Base64 overhead. Check the stats panel before deploying.

Browser & Node.js only

Relies on btoa/atob and pako. Server-side ports for PHP/Python/C# provided in snippets above.

kUrlObfushortner-secure

An encryption layer on top of the base plugin. The browser encrypts the compressed token with your RSA public key — only your server, holding the private key, can decrypt it. Tokens are opaque to anyone without the key.

Secure Plugin — Overview & Architecture

🔒 RSA-OAEP

kUrlObfushortner-secure.js wraps the base plugin and adds RSA-OAEP asymmetric encryption via the browser's native Web Crypto API. The public key lives in the browser; the private key never leaves your server. Decryption must happen server-side.

⚠ RSA-OAEP ciphertext is always the same size as the key modulus (256 bytes for 2048-bit, 512 for 4096-bit) — the encrypted URL will always be longer than the compressed one. The tradeoff is confidentiality: without the private key, the original URL is unrecoverable.

Node.js — Express (Pattern A: URL arrives at server)

Middleware: decrypt _ks on arrival, rewrite req.url

const crypto = require('crypto');
const pako   = require('pako');
const fs     = require('fs');

const privateKeyPem = fs.readFileSync('/etc/secrets/private.pem', 'utf8');

function kurlSecureDecode(encryptedToken) {
  // 1. URL-safe Base64 → Buffer
  const b64      = encryptedToken.replace(/-/g, '+').replace(/_/g, '/');
  const pad      = '='.repeat((4 - b64.length % 4) % 4);
  const cipher   = Buffer.from(b64 + pad, 'base64');

  // 2. RSA-OAEP decrypt
  const plain    = crypto.privateDecrypt(
    { key: privateKeyPem, padding: crypto.constants.RSA_PKCS1_OAEP_PADDING,
      oaepHash: 'sha256' },
    cipher
  );

  // 3. Drop the version byte (0x02), then raw DEFLATE inflate the rest.
  //    plain is raw bytes: [0x02, ...deflateRaw(qs)] — not a _k token.
  const comp     = plain.subarray(1);
  const qs       = pako.inflateRaw(comp, { to: 'string' });
  return qs; // e.g. "?foo=bar&baz=qux"
}

// Express middleware
app.use((req, res, next) => {
  const ks = req.query._ks;
  if (ks) {
    try {
      const qs  = kurlSecureDecode(ks);
      req.url   = req.path + qs;
      req.query = Object.fromEntries(new URLSearchParams(qs));
    } catch (e) { console.error('kurl-secure decode error', e); }
  }
  next();
});

PHP 8+ — Pattern A (direct arrival)

Decrypt _ks in a PHP page/controller on arrival

<?php
// Requires: php-zlib (default) + the openssl CLI on PATH.
// Store your private key outside the web root!
//
// The browser uses RSA-OAEP with SHA-256, but PHP's openssl_private_decrypt()
// only does OAEP with SHA-1 (no way to change it). So we shell out to the
// openssl CLI, which lets us set rsa_oaep_md:sha256. (phpseclib also works.)

function kurlSecureDecode(string $token): string {
  // 1. URL-safe Base64 → binary ciphertext
  $pad    = str_repeat('=', (4 - strlen($token) % 4) % 4);
  $cipher = base64_decode(strtr($token . $pad, '-_', '+/'));

  // 2. RSA-OAEP (SHA-256) decrypt via the openssl CLI; ciphertext via stdin
  $cmd  = ['openssl', 'pkeyutl', '-decrypt',
           '-inkey', '/etc/secrets/private.pem',
           '-pkeyopt', 'rsa_padding_mode:oaep',
           '-pkeyopt', 'rsa_oaep_md:sha256',
           '-pkeyopt', 'rsa_mgf1_md:sha256'];
  $spec = [0 => ['pipe', 'r'], 1 => ['pipe', 'w'], 2 => ['pipe', 'w']];
  $proc = proc_open($cmd, $spec, $pipes);
  fwrite($pipes[0], $cipher);
  fclose($pipes[0]);
  $plain = stream_get_contents($pipes[1]);
  fclose($pipes[1]); fclose($pipes[2]);
  if (proc_close($proc) !== 0) {
    throw new RuntimeException('RSA-OAEP decrypt failed');
  }

  // 3. Drop the version byte (0x02), then raw DEFLATE inflate the rest.
  //    $plain is raw bytes: chr(0x02) . deflateRaw(qs) — not a _k token.
  $qs = gzinflate(substr($plain, 1));

  return $qs; // e.g. "?foo=bar&baz=qux"
}

// Pattern A: use on page arrival
if (!empty($_GET['_ks'])) {
  $qs = kurlSecureDecode($_GET['_ks']);
  parse_str(ltrim($qs, '?'), $_GET);
  // Now $_GET contains the original parameters
}

Python 3 — Flask (Pattern A: direct arrival)

Decrypt _ks in a Flask route

# pip install cryptography flask
import zlib, base64
from cryptography.hazmat.primitives import hashes, serialization
from cryptography.hazmat.primitives.asymmetric import padding
from flask import Flask, request, jsonify

app = Flask(__name__)

with open('/etc/secrets/private.pem', 'rb') as f:
    private_key = serialization.load_pem_private_key(f.read(), password=None)

def kurl_secure_decode(token: str) -> str:
    # 1. URL-safe Base64 → bytes
    pad      = '=' * ((4 - len(token) % 4) % 4)
    cipher   = base64.urlsafe_b64decode(token + pad)

    # 2. RSA-OAEP decrypt
    plain    = private_key.decrypt(
        cipher,
        padding.OAEP(
            mgf=padding.MGF1(algorithm=hashes.SHA256()),
            algorithm=hashes.SHA256(),
            label=None
        )
    )

    # 3. Drop the version byte (0x02), then raw DEFLATE inflate the rest.
    #    plain is raw bytes: bytes([0x02]) + deflate_raw(qs) — not a _k token.
    comp     = plain[1:]
    qs       = zlib.decompress(comp, -15).decode()
    return qs  # e.g. "?foo=bar&baz=qux"

# Pattern A: Flask route that auto-decodes _ks
@app.before_request
def decode_secure_url():
    token = request.args.get('_ks')
    if token:
        from urllib.parse import parse_qs
        qs_str = kurl_secure_decode(token).lstrip('?')
        request.decoded_args = parse_qs(qs_str)

# Pattern B: AJAX decode endpoint
@app.route('/api/kurl-decode', methods=['POST'])
def kurl_decode_endpoint():
    token = request.get_json().get('token')
    if not token:
        return jsonify(error='Missing token'), 400
    try:
        qs = kurl_secure_decode(token)
        return jsonify(queryString=qs, decodedUrl=request.host_url + qs)
    except Exception as e:
        return jsonify(error='Decryption failed'), 400
C# / ASP.NET Core (Pattern A + B)

Middleware + controller endpoint

using System.IO.Compression;
using System.Security.Cryptography;
using System.Text;

public static class KUrlSecure
{
  private static readonly string _privKeyPem =
    File.ReadAllText("/etc/secrets/private.pem");

  public static string Decode(string encryptedToken)
  {
    // 1. URL-safe Base64 → bytes
    var b64    = encryptedToken.Replace('-', '+').Replace('_', '/');
    var pad    = new string('=', (4 - b64.Length % 4) % 4);
    var cipher = Convert.FromBase64String(b64 + pad);

    // 2. RSA-OAEP decrypt
    using var rsa    = RSA.Create();
    rsa.ImportFromPem(_privKeyPem);
    var plainBytes   = rsa.Decrypt(cipher, RSAEncryptionPadding.OaepSHA256);

    // 3. Drop the version byte (0x02), then raw DEFLATE inflate the rest.
    //    plainBytes is raw: [0x02, ...deflateRaw(qs)] — not a _k token.
    var compressed   = plainBytes[1..];

    using var ms     = new MemoryStream(compressed);
    using var ds     = new DeflateStream(ms, CompressionMode.Decompress);
    using var sr     = new StreamReader(ds, Encoding.UTF8);
    return sr.ReadToEnd(); // "?foo=bar&baz=qux"
  }
}

// Pattern A: ASP.NET Core middleware
app.Use(async (context, next) => {
  if (context.Request.Query.ContainsKey("_ks")) {
    var qs = KUrlSecure.Decode(context.Request.Query["_ks"]!);
    context.Request.QueryString = new QueryString(qs);
  }
  await next(context);
});

// Pattern B: minimal API endpoint
app.MapPost("/api/kurl-decode", (DecodeRequest req) => {
  try {
    var qs = KUrlSecure.Decode(req.Token);
    return Results.Ok(new { queryString = qs });
  } catch {
    return Results.BadRequest(new { error = "Decryption failed" });
  }
});
